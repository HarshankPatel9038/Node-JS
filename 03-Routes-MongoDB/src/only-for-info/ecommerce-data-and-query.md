2. Retrieve the list of users who have made more than 2 orders.
[
  {
    $group: {
      _id: "$user_id",
      "count_order": {
        $sum: 1
      },
      order_id: {$push: "$_id"}
    }
  },
  {
    $match: {
      count_order: {$gte: 2}
    }
  },
  {
    $lookup: {
      from: "users",
      localField: "_id",
      foreignField: "_id",
      as: "users"
    }
  },
  {
    $unwind: {
      path: "$users"
    }
  },
  {
    $project: {
      _id: 1,
      order_id: 1,
      count_order: 1,
      name: "$users.name",
      mobile_no: "$users.mobile_no"
    }
  }
]

19. Find the top 3 subcategories with the highest average product price.
[
  {
    $unwind: {
      path: '$attributes',
    }
  },
  {
    $lookup: {
      from: 'products',
      localField: '_id',
      foreignField: '_id',
      as: 'products_result'
    }
  },
  {
    $unwind: {
      path: '$products_result',
    }
  },
  {
    $lookup: {
      from: 'subcategories',
      localField: 'products_result._id',
      foreignField: '_id',
      as: 'subcategory_result'
    }
  },
  {
    $unwind: {
      path: '$subcategory_result',
    }
  },
  {
    $group: {
      _id: '$subcategory_result._id',
      subcategory_name: {
        $first: "$subcategory_result.subcategory_name"
      },
      avg_price: {
        $avg: '$attributes.Price'
      },
      products: {
        "$push": "$products_result.name"
      }
    }
  },
  {
    $sort: {
      avg_price: -1
    }
  },
  {
    $limit: 3
  },
]
